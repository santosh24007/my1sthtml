    <script src="https://sdk.amazonaws.com/js/aws-sdk-2.1023.0.min.js"></script>
    <script>
        // AWS Configuration
        AWS.config.update({
            region: 'us-east-1', // Replace with your S3 bucket's region
            credentials: new AWS.Credentials('AKIAXNGUVGAKOFA73ZFE', 'ExA55Vpxbc8YuZh6cyrw3V1nlbp46e3IN8Gw2NDF')  // Use your own credentials
        });

        const s3 = new AWS.S3();
        const bucketName = 'my-bucket-santosh24007'; // Replace with your bucket name

        // Handle file selection
        document.getElementById('fileInput').addEventListener('change', function () {
            const files = this.files;
            if (files.length > 0) {
                document.getElementById('uploadButton').disabled = false;  // Enable upload button once files are selected
            }
        });

        // Upload event
        document.getElementById('uploadButton').addEventListener('click', function () {
            const files = document.getElementById('fileInput').files;
            const statusMessage = document.getElementById('statusMessage');
            const successPage = document.getElementById('successPage');

            if (files.length === 0) {
                alert('Please select files or folders');
                return;
            }

            // Show status message
            statusMessage.textContent = 'Uploading...';
            statusMessage.style.display = 'block';

            let uploadedFilesCount = 0;
            let totalFiles = files.length;

            // Recursive function to upload files
            function uploadFile(index) {
                const file = files[index];
                const params = {
                    Bucket: bucketName,
                    Key: file.webkitRelativePath || file.name,  // Keep relative path for folders
                    Body: file,
                    ContentType: file.type,
                    StorageClass: 'GLACIER_IR'  // Ensure the storage class is set to Glacier IR
                };

                // Upload the file to S3
                s3.upload(params, function (err, data) {
                    if (err) {
                        console.error('Error uploading file:', err);
                        statusMessage.textContent = 'Error: ' + err.message;
                        statusMessage.style.background = 'rgba(255, 0, 0, 0.8)';
                    } else {
                        uploadedFilesCount++;
                        console.log('Upload success:', data);
                        statusMessage.textContent = `Uploaded ${uploadedFilesCount} of ${totalFiles} files`;
                        statusMessage.style.background = 'rgba(0, 128, 0, 0.8)';

                        // Upload next file
                        if (uploadedFilesCount < totalFiles) {
                            uploadFile(uploadedFilesCount);
                        } else {
                            // Show success page once all files are uploaded
                            successPage.style.display = 'block';
                            statusMessage.style.display = 'none';  // Hide the status message after success
                            document.getElementById('uploadButton').disabled = true; // Disable the upload button after success
                        }
                    }
                });
            }

            // Start uploading the first file
            uploadFile(0);
        });
    </script>
